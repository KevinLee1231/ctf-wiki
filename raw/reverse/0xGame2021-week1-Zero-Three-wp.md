## 题目简述

程序要求输入 32 个可打印字符。前 10 条约束直接作用于前 16 个字节，随后又按小端序把每 4 个字节组成一个 32 位整数，并用 8 条线性方程约束这 8 个整数。把两层约束原样交给 Z3 即可求出唯一输入。

## 解题过程

先为 32 个字符建立整数变量，并限制到可打印 ASCII 范围 $0x20<p_i<127$。组合整数满足 $u_i=p_{4i}+2^8p_{4i+1}+2^{16}p_{4i+2}+2^{24}p_{4i+3}$，这与源码把字符指针转换成 `uint32_t *` 后在小端机器上读取的行为一致。

```python
from z3 import *

p = [Int('x%d' % i) for i in range(0, 32)]
num = [Int('u%d' % i) for i in range(0, 8)]

s = Solver()

for i in range(0, 32):
    s.add(p[i] > 0x20)
    s.add(p[i] < 127)

s.add(2 * p[5] + 8225 - p[0] - 9 * p[3] - p[4] - p[9] + p[10] - 6 * p[11] - p[13] - 3 * p[14] - 5 * p[15] == 5643)
s.add(3 * p[14] + 8 * p[10] - 3 * p[2] - 6 * p[0] + 8225 - p[1] - 2 * p[5] + 3 * p[8] - 8 * p[11] + 4 * p[12] - 6 * p[15] == 6620)
s.add(-5 * p[12] - 7 * p[6] - 3 * p[1] + 8225 - 2 * p[0] - p[2] - 5 * p[3] - 7 * p[4] - 6 * p[5] - 2 * p[8] + 6 * p[13] == 5538)
s.add(2 * p[5] + 2 * p[1] + 8225 - 2 * p[0] - 2 * p[3] + 3 * p[4] - 2 * p[8] - p[10] - p[12] - 2 * p[14] - 2 * p[15] == 7693)
s.add(-6 * p[14] + 8225 - 2 * p[1] - 2 * p[2] - 9 * p[3] + 2 * p[4] - 5 * p[7] + 2 * p[8] - 9 * p[9] - 4 * p[10] - 6 * p[15] == 4735)
s.add(9 * p[14] - 7 * p[10] + 8 * p[9] + 5 * p[0] + 8225 - p[2] + p[5] - 5 * p[6] - 8 * p[11] - p[12] - 9 * p[15] == 7060)
s.add(p[13] - 5 * p[7] - 3 * p[2] - 3 * p[0] + 8225 - 4 * p[1] - 4 * p[4] - p[6] + 9 * p[10] - 2 * p[14] - 6 * p[15] == 5864)
s.add(-9 * p[14] - 3 * p[10] + 9 * p[1] - 6 * p[0] + 8225 - 5 * p[3] - 4 * p[7] - 2 * p[11] - 2 * p[12] + p[13] + 9 * p[15] == 7393)
s.add(6 * p[9] - 5 * p[8] - 3 * p[6] + 9 * p[2] + 8225 - p[4] + 3 * p[5] - 7 * p[7] + 7 * p[10] - 2 * p[13] - p[14] == 8442)
s.add(8 * p[6] - 7 * p[2] + 8225 - 8 * p[1] - p[3] + 6 * p[4] - p[7] + 5 * p[8] - 4 * p[10] - p[14] + 7 * p[15] == 8376)

for i in range(0, 32, 4):
    s.add(num[i // 4] == (p[i] + (p[i + 1] * 256) + (p[i + 2] * 65536) + (p[i + 3] * 16777216)))

s.add(-22827 * num[4] + 21984 * num[1] - 38534 * num[5] - 32344 * num[0] == -98460819657603)
s.add(-38215 * num[2] - 37324 * num[4] - 8436 * num[5] + 15405 * num[0] == -131665436206262)
s.add(10926 * num[7] - 28942 * num[1] - 34572 * num[3] - 10538 * num[5] == -121891239772992)
s.add(-30117 * num[6] - 22990 * num[2] - 20471 * num[5] + 34494 * num[7] == -57089882568260)
s.add(-33112 * num[5] - 19335 * num[4] + 34348 * num[1] + 31445 * num[2] == 56335531538050)
s.add(-13566 * num[5] + 14758 * num[0] - 19814 * num[2] - 26447 * num[4] == -81105980248303)
s.add(25898 * num[5] - 15817 * num[1] + 20463 * num[7] - 33578 * num[0] == -28860618440412)
s.add(-35429 * num[7] + 36594 * num[2] - 28801 * num[6] - 14952 * num[3] == -45384029412201)

print(s.check())
if s.check() == sat:
    result = s.model()
    raw = b"".join(
        result.eval(num[i]).as_long().to_bytes(4, "little")
        for i in range(8)
    )
    print(f"0xGame{{{raw.decode()}}}")

```

## 方法总结

本题的难点是正确保留两种数据视图：同一段输入既按单字节参与方程，也按每 4 字节小端整数参与方程。Z3 负责求解，人工分析负责保证变量范围、字节序和打包关系与原程序一致；这三项任一写错都会得到无解或错误字符串。
